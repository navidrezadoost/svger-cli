**Implementation:**

```typescript
public generateComponentName(fileName: string): string {
  const baseName = path.basename(fileName, '.svg');
  const componentName = toPascalCase(baseName);

  // Ensure valid React component name
  if (!/^[A-Z]/.test(componentName)) {
    return `Svg${componentName}`;
  }

  return componentName;
}
```

## üöÄ Usage Examples

### CLI Usage

```bash
# Basic usage with defaults (React + TypeScript)
svger-cli build ./svgs ./components

# Specify framework
svger-cli build ./svgs ./components --framework vue

# Use TypeScript/JavaScript
svger-cli build ./svgs ./components --typescript
svger-cli build ./svgs ./components --no-typescript

# Framework-specific options
svger-cli build ./svgs ./components --framework vue --composition
svger-cli build ./svgs ./components --framework angular --standalone

# Watch mode
svger-cli watch ./svgs ./components

# Generate single file
svger-cli generate ./svgs/icon.svg ./components
```

### Programmatic API

```typescript
import { svgService, svgProcessor, configService, toPascalCase, toKebabCase } from 'svger-cli';

// Build all SVGs
await svgService.buildAll({
  src: './svgs',
  out: './components',
  config: {
    framework: 'react',
    typescript: true,
  },
});

// Process single file with naming
const componentName = toPascalCase('icon-home'); // ‚Üí 'IconHome'
await svgProcessor.processSVGFile('./svgs/icon-home.svg', './components', {
  framework: 'react',
  typescript: true,
  namingConvention: 'pascal',
});

// Configuration
configService.setConfig('defaultWidth', 32);
const config = configService.readConfig();
```

### Import Generated Components

```typescript
// Named imports (tree-shaking)
import { IconHome, IconSettings } from './components';

// Namespace import
import * as Icons from './components';
<Icons.IconHome />

// Default import
import Components from './components';
const { IconHome } = Components;

// Dynamic import
const IconHome = await import('./components').then(m => m.IconHome);
```
